58      @Test
59      public void testRead() throws Exception {
60  
61          BeanMapping<UserCsv> beanMapping = beanMappingFactory.create(UserCsv.class);
62  
63          String csv = "id,名前,ホームページ,ホームページ（予備）,E-mail"
64                  + "\r\n" + "001,admin,http://mysite.com/admin.html,https://localhost/sub.html,admin@mysite.com"
65                  + "\r\n" + "002,sample,http://mysite.com/sample.html,https://localhost/sub.html,sample@mysite.com"
66                  + "\r\n" + "003,test,ftp://mysite.com/test.html,https://localhost/sub.html,test@mysite.com"
67                  + "\r\n" + "004,test,http://mysite.com/test.html,abc,test@mysite.com"
68                  + "\r\n" + "005,admin,http://mysite.com/admin.html,https://localhost/sub.html,admin_mysite.com"
69                  ;
70  
71          CsvAnnotationBeanReader<UserCsv> csvReader = new CsvAnnotationBeanReader<>(beanMapping,
72                  new StringReader(csv), CsvPreference.STANDARD_PREFERENCE);
73          csvReader.setExceptionConverter(exceptionConverter);
74  
75          List<UserCsv> list = csvReader.readAll(true);
76          csvReader.close();
77  
78          csvReader.getErrorMessages().stream().forEach(System.out::println);
79  
80          assertThat(csvReader.getErrorMessages()).hasSize(4)
81              .contains("[3行, 2列] : 項目「名前」の値（sample）は、存在しません。")
82              .contains("[4行, 3列] : 項目「ホームページ」の値（ftp://mysite.com/test.html）のプロトコル「ftp」はサポートしていません。")
83              .contains("[5行, 4列] : 項目「ホームページ（予備）」の値（abc）は、サブのホームページの値として不正です。")
84              .contains("[6行, 5列] : 項目「E-mail」の値（admin_mysite.com）は、メールアドレスの書式として不正です。")
85              ;
86  
87          assertThat(list).hasSize(1);
88          assertThat(list.get(0)).hasFieldOrPropertyWithValue("id", "001");
89  
90      }
